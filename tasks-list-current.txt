function getField(key: string) {
  return fieldDefinitions.find((field) => field.key === key);
}

function getOptionDescription(
  options: OptionDefinition[] | undefined,
  value: string | number | boolean | null | undefined
) {
  if (!options) return "";
  return (
    options.find((option) => option.value === String(value))?.description || ""
  );
}

function valueToNumberOrNull(
  value: string | number | boolean | null | undefined
) {
  if (value === "" || value === null || value === undefined) return null;
  return Number(value);
}

function valueToBoolean(value: string | number | boolean | null | undefined) {
  return value === true || value === "true" || value === "Yes" || value === "1";
}

function booleanToYesNo(value: string | number | boolean | null | undefined) {
  return valueToBoolean(value) ? "Yes" : "No";
}

function formatDate(value: string | number | boolean | null | undefined) {
  if (!value) return "-";
  return String(value).split("T")[0];
}

function formatTime(value: string | number | boolean | null | undefined) {
  if (!value) return "-";
  return String(value).slice(0, 5);
}

function formatDateLabel(value: string | number | boolean | null | undefined) {
  if (!value) return "Select";

  const date = new Date(`${String(value).split("T")[0]}T12:00:00`);

  return date.toLocaleDateString("en-US", {
    month: "short",
    day: "numeric",
  });
}

function formatMonthYear(date: Date) {
  return date.toLocaleDateString("en-US", {
    month: "long",
    year: "numeric",
  });
}

function toDateInputValue(date: Date) {
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, "0");
  const day = String(date.getDate()).padStart(2, "0");

  return `${year}-${month}-${day}`;
}

function getMonthDays(calendarDate: Date) {
  const year = calendarDate.getFullYear();
  const month = calendarDate.getMonth();
  const firstDay = new Date(year, month, 1);
  const firstWeekday = firstDay.getDay();
  const daysInMonth = new Date(year, month + 1, 0).getDate();
  const cells: Array<number | null> = [];

  for (let index = 0; index < firstWeekday; index += 1) {
    cells.push(null);
  }

  for (let day = 1; day <= daysInMonth; day += 1) {
    cells.push(day);
  }

  while (cells.length % 7 !== 0) {
    cells.push(null);
  }

  return cells;
}

function escapeCsvValue(value: string | number | boolean | null | undefined) {
  if (value === null || value === undefined) return "";
  const text = String(value);

  if (text.includes(",") || text.includes('"') || text.includes("\n")) {
    return `"${text.replace(/"/g, '""')}"`;
  }

  return text;
}

function parseCsvLine(line: string) {
  const values: string[] = [];
  let current = "";
  let inQuotes = false;

  for (let index = 0; index < line.length; index += 1) {
    const character = line[index];
    const nextCharacter = line[index + 1];

    if (character === '"' && inQuotes && nextCharacter === '"') {
      current += '"';
      index += 1;
      continue;
    }

    if (character === '"') {
      inQuotes = !inQuotes;
      continue;
    }

    if (character === "," && !inQuotes) {
      values.push(current);
      current = "";
      continue;
    }

    current += character;
  }

  values.push(current);
  return values;
}

function parseCsv(text: string) {
  const lines = text
    .split(/\r?\n/)
    .map((line) => line.trimEnd())
    .filter((line) => line.trim() !== "");

  if (lines.length < 2) {
    return { headers: [] as string[], rows: [] as Record<string, string>[] };
  }

  const headers = parseCsvLine(lines[0]).map((header) => header.trim());

  const rows = lines.slice(1).map((line) => {
    const values = parseCsvLine(line);
    const row: Record<string, string> = {};

    headers.forEach((header, index) => {
      row[header] = values[index] || "";
    });

    return row;
  });

  return { headers, rows };
}

function isOverdue(row: RowData) {
  if (!row.due_date) return false;
  if (["Complete", "Canceled"].includes(String(row.status || ""))) return false;

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const due = new Date(String(row.due_date));
  due.setHours(0, 0, 0, 0);

  return due < today;
}

function isDueSoon(row: RowData) {
  if (!row.due_date) return false;
  if (["Complete", "Canceled"].includes(String(row.status || ""))) return false;

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const due = new Date(String(row.due_date));
  due.setHours(0, 0, 0, 0);

  const diffDays = Math.ceil(
    (due.getTime() - today.getTime()) / (1000 * 60 * 60 * 24)
  );

  return diffDays >= 0 && diffDays <= 7;
}

      setDeliverables((deliverablesResult.data as LinkedDeliverable[]) || []);
    }

    if (!issuesResult.error) {
      setIssues((issuesResult.data as LinkedIssue[]) || []);
    }

    setLoading(false);
  }

  function getEmployeeName(id: string | number | boolean | null | undefined) {
    if (!id) return "-";
    const employee = employees.find((item) => item.id === Number(id));
    return employee?.name || `Employee #${id}`;
  }

  function displayValue(row: RowData, key: string) {
    if (key === "assigned_to") return getEmployeeName(row[key]);
    if (key === "due_date") return formatDate(row[key]);
    if (key === "due_time") return formatTime(row[key]);
    if (key === "skip_weekends") return booleanToYesNo(row[key]);
    return String(row[key] ?? "-");
  }

  function updateDraft(key: string, value: string | number | boolean | null) {
    setDraft((current) => ({
      ...current,
      [key]: value,
    }));
  }

  function cleanPayload(source: RowData) {
    const isRepeating = String(source.due_date_type || "Standard") === "Repeating";

    return {
      task_title: source.task_title || null,
      status: source.status || "To Do",
      priority: source.priority || "Important",
      assigned_to: valueToNumberOrNull(source.assigned_to),
      task_type: source.task_type || null,
      function_area: source.function_area || null,

      due_date_type: isRepeating ? "Repeating" : "Standard",
      due_date: source.due_date || null,
      due_time: source.due_time || null,

      recurrence_frequency: isRepeating
        ? source.recurrence_frequency || null
        : null,
      recurrence_interval: isRepeating
        ? Number(source.recurrence_interval || 1)
        : 1,
      recurrence_days: isRepeating ? source.recurrence_days || null : null,
      recurrence_monthly_rule: isRepeating
        ? source.recurrence_monthly_rule || source.recurrence_days || null
        : null,
      skip_weekends: isRepeating ? valueToBoolean(source.skip_weekends) : false,
      create_next_when: isRepeating
        ? source.create_next_when || "Complete"
        : null,
      recurrence_end_date: isRepeating
        ? source.recurrence_end_date || null
        : null,
      next_task_status: isRepeating ? source.next_task_status || "To Do" : null,
      recurrence_fields_to_copy: isRepeating
        ? source.recurrence_fields_to_copy || "All Fields"
        : null,

      ai_prompt_tool_link: source.ai_prompt_tool_link || defaultAiToolLink,
      description: source.description || null,
      notes: source.notes || null,
    };
  }

  async function saveRecord() {
    setMessage("");
    setErrorMessage("");

    if (!String(draft.task_title || "").trim()) {
      setErrorMessage("Task Title is required.");
      return;
    }

    const supabase = createClient();

    if (modalMode === "add") {
      const { error } = await supabase.from("tasks").insert(cleanPayload(draft));

      if (error) {
        setErrorMessage(error.message);
        return;
      }

      setMessage("Task added.");
    }

    if (modalMode === "edit" && activeRecord?.id) {
      const { error } = await supabase
        .from("tasks")
        .update(cleanPayload(draft))
        .eq("id", activeRecord.id);

      if (error) {
        setErrorMessage(error.message);
        return;
      }

      setMessage("Task updated.");
    }

    closeModal();
    await loadEverything();
  }

  function closeModal() {
    setModalMode(null);
    setActiveRecord(null);
    setDraft({ ...initialDraft });
  }

  async function duplicateRecord(row: RowData) {
    setMessage("");
    setErrorMessage("");

    const copy = { ...row };
    delete copy.id;
    delete copy.created_at;

    copy.task_title = `${row.task_title || "Task"} Copy`;
    copy.status = "To Do";

    const supabase = createClient();
    const { error } = await supabase.from("tasks").insert(cleanPayload(copy));

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage("Task duplicated.");
    setRowActionMenuId(null);
    await loadEverything();
  }

  async function deleteRecord(row: RowData) {
    const taskId = Number(row.id);
    const linkedDeliverables = deliverables.filter(
      (item) => Number(item.linked_task) === taskId
    );
    const linkedIssues = issues.filter(
      (item) => Number(item.linked_task) === taskId
    );

    if (linkedDeliverables.length > 0 || linkedIssues.length > 0) {
      setErrorMessage(
        `This task cannot be deleted because it has ${linkedDeliverables.length} linked deliverable(s) and ${linkedIssues.length} linked issue(s). Open the task and unlink those records first.`
      );
      setRowActionMenuId(null);
      return;
    }

    const confirmed = window.confirm(
      `Delete "${row.task_title || "this task"}"? This cannot be undone.`
    );

    if (!confirmed) return;

    const supabase = createClient();
    const { error } = await supabase.from("tasks").delete().eq("id", row.id);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage("Task deleted.");
    setSelectedRowIds((current) =>
      current.filter((id) => id !== String(row.id))
    );
    setRowActionMenuId(null);
    await loadEverything();
  }

  function startCellEdit(row: RowData, field: FieldDefinition) {
    setActiveCell({
      rowId: String(row.id),
      fieldKey: field.key,
    });

    if (field.type === "boolean") {
      setCellValue(booleanToYesNo(row[field.key]));
    } else {
      setCellValue(String(row[field.key] ?? ""));
    }
  }

  async function saveCellEdit(row: RowData, field: FieldDefinition) {
    const supabase = createClient();

    let value: string | number | boolean | null = cellValue;

    if (field.type === "number") value = valueToNumberOrNull(cellValue);
    if (field.type === "employee") value = valueToNumberOrNull(cellValue);
    if (field.type === "boolean") value = cellValue === "Yes";
    if (field.type === "date" && !cellValue) value = null;
    if (field.type === "time" && !cellValue) value = null;

    const { error } = await supabase
      .from("tasks")
      .update({ [field.key]: value })
      .eq("id", row.id);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setRows((current) =>
      current.map((item) =>
        String(item.id) === String(row.id)
          ? { ...item, [field.key]: value }
          : item
      )
    );

    setActiveCell(null);
    setCellValue("");
    setMessage("Cell updated.");
  }

  function renderSmallInput(
    field: FieldDefinition,
    value: string,
    setValue: (value: string) => void

              key={option.value}
              value={option.value}
              title={option.description}

              {option.label}
            </option>
          ))}
        </select>
      );

    if (field.type === "employee") {
      return (
        <select
          value={value}
          onChange={(event) => setValue(event.target.value)}
          style={inlineInputStyle}
        >
          <option value="">Select employee</option>

            <option key={employee.id} value={employee.id}>
              {employee.name || `Employee #${employee.id}`}
            </option>

    return (
      <input
        type={
          field.type === "date" ? "date" : field.type === "time" ? "time" : "text"

        value={value}
        onChange={(event) => setValue(event.target.value)}
        style={inlineInputStyle}
      />
    );
  }

  async function bulkUpdate() {
    if (selectedRowIds.length === 0) return;

    const field = getField(bulkUpdateFieldKey);
    if (!field) return;

    let value: string | number | boolean | null = bulkUpdateValue;

    if (field.type === "employee" || field.type === "number") {
      value = valueToNumberOrNull(bulkUpdateValue);
    }

    if (field.type === "boolean") {
      value = bulkUpdateValue === "Yes";
    }

    if ((field.type === "date" || field.type === "time") && !bulkUpdateValue) {
      value = null;
    }

    const supabase = createClient();

    const { error } = await supabase
      .from("tasks")
      .update({ [bulkUpdateFieldKey]: value })
      .in("id", selectedRowIds);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage("Selected tasks updated.");
    setSelectedRowIds([]);
    await loadEverything();
  }

  async function bulkDuplicate() {
    if (selectedRows.length === 0) return;

    const copies = selectedRows.map((row) => {
      const copy = { ...row };
      delete copy.id;
      delete copy.created_at;

      copy.task_title = `${row.task_title || "Task"} Copy`;
      copy.status = "To Do";

      return cleanPayload(copy);
    });

    const supabase = createClient();
    const { error } = await supabase.from("tasks").insert(copies);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage("Selected tasks duplicated.");
    setSelectedRowIds([]);
    await loadEverything();
  }

  async function bulkDelete() {
    if (selectedRows.length === 0) return;

    const blockedRows = selectedRows.filter((row) => {
      const taskId = Number(row.id);

      return (
        deliverables.some((item) => Number(item.linked_task) === taskId) ||
        issues.some((item) => Number(item.linked_task) === taskId)
      );
    });

    if (blockedRows.length > 0) {
      setErrorMessage(
        `${blockedRows.length} selected task(s) have linked deliverables or issues. Unlink those records before deleting.`
      );
      return;
    }

    const confirmed = window.confirm(
      `Delete ${selectedRows.length} selected task(s)? This cannot be undone.`
    );

    if (!confirmed) return;

    const supabase = createClient();

    const { error } = await supabase
      .from("tasks")
      .delete()
      .in("id", selectedRowIds);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage("Selected tasks deleted.");
    setSelectedRowIds([]);
    await loadEverything();
  }

  function exportCsv() {
    const headers = fieldDefinitions.map((field) => field.key);
    const csvRows = [
      headers.join(","),
      ...processedRows.map((row) =>
        headers.map((header) => escapeCsvValue(row[header])).join(",")
      ),
    ];

    const blob = new Blob([csvRows.join("\n")], {
      type: "text/csv;charset=utf-8;",
    });

    const url = URL.createObjectURL(blob);
    const link = document.createElement("a");

    link.href = url;
    link.download = "tasks-export.csv";
    link.click();

    URL.revokeObjectURL(url);
  }

  async function importCsv(file: File) {
    const text = await file.text();
    const parsed = parseCsv(text);

    if (parsed.rows.length === 0) {
      setErrorMessage("No rows found in the CSV file.");
      return;
    }

    const payloads = parsed.rows.map((row) =>
      cleanPayload({
        ...initialDraft,
        ...row,
        assigned_to: row.assigned_to || null,
        recurrence_interval: row.recurrence_interval
          ? Number(row.recurrence_interval)
          : 1,
        skip_weekends: valueToBoolean(row.skip_weekends),
      })
    );

    const supabase = createClient();

    const { error } = await supabase.from("tasks").insert(payloads);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage(`${payloads.length} task(s) imported.`);
    await loadEverything();
  }

  function changeVisibleField(key: string) {
    setVisibleFieldKeys((current) => {
      if (current.includes(key)) {
        if (current.length === 1) return current;
        return current.filter((item) => item !== key);
      }

      return [...current, key];
    });
  }

  function moveVisibleField(key: string, direction: "up" | "down") {
    setVisibleFieldKeys((current) => {
      const index = current.indexOf(key);

      if (index === -1) return current;

      const nextIndex = direction === "up" ? index - 1 : index + 1;

      if (nextIndex < 0 || nextIndex >= current.length) return current;

      const copy = [...current];
      const item = copy[index];

      copy[index] = copy[nextIndex];
      copy[nextIndex] = item;

      return copy;
    });
  }

                          ? selectedFieldButtonStyle
                          : fieldButtonStyle
                      }
                    >
                      {visibleFieldKeys.includes(field.key) ? "✓ " : "+ "}
                      {field.label}
                    </button>
                  ))}
              </div>

              <div style={viewColumnStyle}>
                <h3 style={viewHeadingStyle}>
                  Selected Fields ({visibleFieldKeys.length})
                </h3>
                <p style={viewHelpStyle}>
                  Reorder or remove the fields shown in the table.
                </p>

                {visibleFieldKeys.map((key) => {
                  const field = getField(key);

                  if (!field) return null;

                  return (
                    <div key={key} style={selectedFieldRowStyle}>
                      <span title={field.description}>{field.label}</span>

                      <div style={selectedFieldControlsStyle}>
                        <button
                          type="button"
                          onClick={() => moveVisibleField(key, "up")}
                          style={tinyButtonStyle}
                        >
                          ↑
                        </button>

                        <button
                          type="button"
                          onClick={() => moveVisibleField(key, "down")}
                          style={tinyButtonStyle}
                        >
                          ↓
                        </button>

                        <button
                          type="button"
                          onClick={() => changeVisibleField(key)}
                          style={tinyDangerButtonStyle}
                        >
                          Remove
                        </button>
                      </div>
                    </div>
                  );
                })}
              </div>
            </div>
          </div>
        ) : null}

        {selectedRowIds.length > 0 ? (
          <div style={bulkToolbarStyle}>
            <div style={bulkTopRowStyle}>
              <strong>{selectedRowIds.length} selected</strong>

              <div style={bulkButtonGroupStyle}>
                <button
                  type="button"
                  onClick={bulkDuplicate}
                  style={secondaryButtonStyle}
                >
                  Bulk Duplicate
                </button>

                <button
                  type="button"
                  onClick={bulkDelete}
                  style={dangerButtonStyle}
                >
                  Bulk Delete
                </button>
              </div>
            </div>

            <div style={bulkUpdateGridStyle}>
              <FieldSelect
                label="Bulk Update Field"
                value={bulkUpdateFieldKey}
                options={fieldDefinitions
                  .filter((field) => field.bulk)
                  .map((field) => field.key)}
                onChange={(value) => {
                  setBulkUpdateFieldKey(value);

                  const field = getField(value);

                  if (field?.options?.[0]) {
                    setBulkUpdateValue(field.options[0].value);
                  } else {
                    setBulkUpdateValue("");
                  }
                }}
              />

              <label style={fieldStyle}>
                New Value
                <small style={helpTextStyle}>Value applied to selected tasks.</small>
                {renderBulkValueInput()}
              </label>

              <button type="button" onClick={bulkUpdate} style={primaryButtonStyle}>
                Apply Bulk Update
              </button>
            </div>
          </div>
        ) : null}

        {loading ? (
          <div style={loadingStyle}>Loading tasks...</div>
        ) : (
          <div style={tableWrapStyle}>
            <table style={tableStyle}>
              <thead>
                <tr>
                  <th style={checkboxHeaderStyle}>
                    <input
                      type="checkbox"
                      checked={allVisibleSelected}
                      onChange={toggleAllVisibleRows}
                    />
                  </th>

                  {visibleFields.map((field) => (
                    <th
                      key={field.key}
                      style={thStyle}
                      title={`${field.label}: ${field.description}`}
                    >
                      {field.label}
                    </th>
                  ))}

                  <th style={actionsThStyle}>Actions</th>
                </tr>
              </thead>

              <tbody>
                {groupedRows.map((group) => (
                  <FragmentGroup
                    key={group.group || "all"}
                    groupLabel={group.group}
                    rows={group.rows}
                    visibleFields={visibleFields}
                    selectedRowIds={selectedRowIds}
                    rowActionMenuId={rowActionMenuId}
                    setRowActionMenuId={setRowActionMenuId}
                    toggleRowSelection={toggleRowSelection}
                    renderInlineEditor={renderInlineEditor}
                    openRecord={openRecord}
                    editRecord={editRecord}
                    duplicateRecord={duplicateRecord}
                    deleteRecord={deleteRecord}
                  />
                ))}

                {processedRows.length === 0 ? (
                  <tr>
                    <td colSpan={visibleFields.length + 2} style={emptyStateStyle}>
                      No tasks found. Use + Add Task to create one.
                    </td>
                  </tr>
                ) : null}
              </tbody>
            </table>
          </div>
        )}
      </section>

      {renderModal()}
    </main>
  );

  function renderFilterValueInput() {
    const field = getField(filterFieldKey);

    if (!field) {
      return (
        <input
          value={filterValue}
          onChange={(event) => setFilterValue(event.target.value)}
          style={inputStyle}
        />
      );
    }

    return renderSmallInput(field, filterValue, setFilterValue);
  }

  function renderBulkValueInput() {
    const field = getField(bulkUpdateFieldKey);

    if (!field) {
      return (
        <input
          value={bulkUpdateValue}
          onChange={(event) => setBulkUpdateValue(event.target.value)}
          style={inputStyle}
        />
      );
    }

    return renderSmallInput(field, bulkUpdateValue, setBulkUpdateValue);
  }
}

function DueDateRepeatEditor({
  draft,
  readOnly,
  onChange,
}: {
  draft: RowData;
  readOnly: boolean;
  onChange: (key: string, value: string | number | boolean | null) => void;
}) {
  const [popoverOpen, setPopoverOpen] = useState(false);
  const [calendarDate, setCalendarDate] = useState(() => {
    if (draft.due_date) {
      return new Date(`${String(draft.due_date).split("T")[0]}T12:00:00`);
    }

    return new Date();
  });

  const calendarDays = useMemo(() => getMonthDays(calendarDate), [calendarDate]);
  const isRepeating = String(draft.due_date_type || "Standard") === "Repeating";

  if (readOnly) {
    return (
      <div style={linkedGridStyle}>
        <div style={infoBoxStyle}>
          <div style={infoLabelStyle}>Due Date</div>
          <div style={infoValueStyle}>{formatDate(draft.due_date)}</div>
          <small style={infoHelperStyle}>The date the task should be completed.</small>
        </div>

        <div style={infoBoxStyle}>
          <div style={infoLabelStyle}>Due Time</div>
          <div style={infoValueStyle}>{formatTime(draft.due_time)}</div>
          <small style={infoHelperStyle}>Optional time the task should be completed.</small>
        </div>

        <div style={infoBoxStyle}>
          <div style={infoLabelStyle}>Repeat</div>
          <div style={infoValueStyle}>{String(draft.due_date_type || "Standard")}</div>
          <small style={infoHelperStyle}>
            {isRepeating
              ? `${draft.recurrence_frequency || "Monthly"} · ${draft.recurrence_days || "No rule"}`
              : "This task does not repeat."}
          </small>
        </div>
      </div>
    );
  }

  function moveMonth(direction: "previous" | "next") {
    setCalendarDate((current) => {
      const next = new Date(current);
      next.setMonth(current.getMonth() + (direction === "next" ? 1 : -1));
      return next;
    });
  }

  function chooseCalendarDay(day: number) {
    const selectedDate = new Date(
      calendarDate.getFullYear(),
      calendarDate.getMonth(),
      day
    );

    const value = toDateInputValue(selectedDate);

    onChange("due_date", value);

    if (!draft.recurrence_end_date) {
      onChange("recurrence_end_date", value);
    }
  }

  function toggleRepeating() {
    if (isRepeating) {
      onChange("due_date_type", "Standard");
      return;
    }

    onChange("due_date_type", "Repeating");
  }

  return (
    <div style={managerDateFieldWrapStyle}>
      <button
        type="button"
        onClick={() => setPopoverOpen(true)}
        style={managerDateButtonStyle}
      >
        <span>
          <strong>Due Date</strong>
          <small style={managerDateHelpStyle}>
            Click to open calendar and repeat settings.
          </small>
        </span>

        <span style={managerDateBadgeStyle}>
          {draft.due_date ? formatDateLabel(draft.due_date) : "Choose Date"}
        </span>
      </button>

      <label style={managerTimeFieldStyle}>
        Due Time
        <small style={helpTextStyle}>
          Optional time the task should be completed by.
        </small>
        <input
          type="time"
          value={String(draft.due_time || "")}
          onChange={(event) => onChange("due_time", event.target.value)}
          style={inputStyle}
        />
      </label>

      {popoverOpen ? (
        <div style={managerDateBackdropStyle}>
          <div style={managerDatePopoverStyle}>
            <aside style={managerRepeatSideStyle}>
              <div style={managerRepeatHeaderStyle}>
                <h3 style={managerRepeatTitleStyle}>Repeating Task</h3>

                <button
                  type="button"
                  onClick={toggleRepeating}
                  style={
                    isRepeating
                      ? managerSwitchActiveStyle
                      : managerSwitchStyle
                  }
                  aria-label="Toggle repeating task"
                >
                  <span
                    style={
                      isRepeating
                        ? managerSwitchKnobActiveStyle
                        : managerSwitchKnobStyle
                    }
                  />
                </button>
              </div>

              <div style={managerRepeatRowStyle}>
                <label style={managerCompactLabelStyle}>
                  Repeats
                  <span
                    title="How often this task should repeat."
                    style={managerSmallInfoIconStyle}
                  >
                    i
                  </span>
                </label>

                <select
                  value={String(draft.recurrence_frequency || "Monthly")}
                  onChange={(event) =>
                    onChange("recurrence_frequency", event.target.value)
                  }
                  disabled={!isRepeating}
                  style={managerCompactSelectStyle}
                >
                  {recurrenceFrequencyOptions.map((option) => (
                    <option key={option.value} value={option.value}>
                      {option.label}
                    </option>
                  ))}
                </select>

                <small style={managerSmallHelpStyle}>
                  {getOptionDescription(
                    recurrenceFrequencyOptions,
                    draft.recurrence_frequency || "Monthly"
                  )}
                </small>
              </div>

              <div style={managerRepeatRowStyle}>
                <label style={managerCompactLabelStyle}>Repeat Rule</label>

                <select
                  value={String(draft.recurrence_days || "First Friday")}
                  onChange={(event) => {
                    onChange("recurrence_days", event.target.value);
                    onChange("recurrence_monthly_rule", event.target.value);
                  }}
                  disabled={!isRepeating}
                  style={managerCompactSelectStyle}
                >
                  {recurrenceRuleOptions.map((option) => (
                    <option key={option.value} value={option.value}>
                      {option.label}
                    </option>
                  ))}
                </select>

                <small style={managerSmallHelpStyle}>
                  {getOptionDescription(
                    recurrenceRuleOptions,
                    draft.recurrence_days || "First Friday"
                  )}
                </small>
              </div>

              <div style={managerDividerStyle} />

              <div style={managerRepeatRowStyle}>
                <label style={managerCompactLabelStyle}>
                  Create new task
                  <small style={managerMutedSmallStyle}>
                    when status is set to completed
                  </small>
                </label>

                <select
                  value={String(draft.create_next_when || "Complete")}
                  onChange={(event) =>
                    onChange("create_next_when", event.target.value)
                  }
                  disabled={!isRepeating}
                  style={managerCompactSelectStyle}
                >
                  {createNextWhenOptions.map((option) => (
                    <option key={option.value} value={option.value}>
                      {option.label}
                    </option>
                  ))}
                </select>
              </div>

              <div style={managerDividerStyle} />

              <div style={managerRepeatInlineRowStyle}>
                <span style={managerCompactLabelStyle}>Set Status to</span>

                <select
                  value={String(draft.next_task_status || "To Do")}
                  onChange={(event) =>
                    onChange("next_task_status", event.target.value)
                  }
                  disabled={!isRepeating}
                  style={managerStatusPillSelectStyle}
                >
                  {statusOptions.map((option) => (
                    <option key={option.value} value={option.value}>
                      {option.label}
                    </option>
                  ))}
                </select>
              </div>

              <div style={managerDividerStyle} />

              <div style={managerRepeatInlineRowStyle}>
                <span style={managerCompactLabelStyle}>Fields to Copy</span>

                <select
                  value={String(draft.recurrence_fields_to_copy || "All Fields")}
                  onChange={(event) =>
                    onChange("recurrence_fields_to_copy", event.target.value)
                  }
                  disabled={!isRepeating}
                  style={managerPlainSelectStyle}
                >
                  {copyFieldsOptions.map((option) => (
                    <option key={option.value} value={option.value}>
                      {option.label}
                    </option>
                  ))}
                </select>
              </div>
            </aside>

            <section style={managerCalendarSideStyle}>
              <div style={managerCalendarTopStyle}>
                <div style={managerDateHeaderBlockStyle}>
                  <span style={managerDateHeaderLabelStyle}>Start Date</span>
                  <strong style={managerDateHeaderValueStyle}>
                    {formatDateLabel(draft.due_date)}
                  </strong>
                </div>

                <div style={managerDateHeaderBlockStyle}>
                  <span style={managerDateHeaderLabelStyle}>End Date</span>
                  <strong style={managerDateHeaderValueStyle}>
                    {formatDateLabel(draft.recurrence_end_date || draft.due_date)}
                  </strong>
                </div>
              </div>

              <div style={managerCalendarNavStyle}>
                <button
                  type="button"
                  onClick={() => moveMonth("previous")}
                  style={managerCalendarArrowStyle}
                >
                  ←
                </button>

                <strong>{formatMonthYear(calendarDate)}</strong>

                <button
                  type="button"
                  onClick={() => moveMonth("next")}
                  style={managerCalendarArrowStyle}
                >
                  →
                </button>
              </div>

              <div style={managerCalendarGridStyle}>
                {["Su", "Mo", "Tu", "We", "Th", "Fr", "Sa"].map((day) => (
                  <div key={day} style={managerWeekdayStyle}>
                    {day}
                  </div>
                ))}

                {calendarDays.map((day, index) => {
                  const selectedDate =
                    day !== null
                      ? toDateInputValue(
                          new Date(
                            calendarDate.getFullYear(),
                            calendarDate.getMonth(),
                            day
                          )
                        )
                      : "";

                  const isSelected = selectedDate === String(draft.due_date || "");

                  return (
                    <button
                      key={`${day || "blank"}-${index}`}
                      type="button"
                      disabled={day === null}
                      onClick={() => {
                        if (day !== null) chooseCalendarDay(day);
                      }}
                      style={
                        day === null
                          ? managerBlankDayStyle
                          : isSelected
                            ? managerSelectedDayStyle
                            : managerDayButtonStyle
                      }
                    >
                      {day || ""}
                    </button>
                  );
                })}
              </div>

              <div style={managerCalendarFooterStyle}>
                <button
                  type="button"
                  style={managerIconButtonStyle}
                  title="Repeat settings"
                  onClick={() => onChange("due_date_type", "Repeating")}
                >
                  ↻
                </button>

                <div style={{ flex: 1 }} />

                <button
                  type="button"
                  onClick={() => setPopoverOpen(false)}
                  style={managerCalendarCancelButtonStyle}
                >
                  Cancel
                </button>

                <button
                  type="button"
                  onClick={() => setPopoverOpen(false)}
                  style={managerCalendarDoneButtonStyle}
                >
                  Done
                </button>
              </div>
            </section>
          </div>
        </div>
      ) : null}
    </div>
  );
}

function FragmentGroup({
  groupLabel,
  rows,
  visibleFields,
  selectedRowIds,
  rowActionMenuId,
  setRowActionMenuId,
  toggleRowSelection,
  renderInlineEditor,
  openRecord,
  editRecord,
  duplicateRecord,
  deleteRecord,
}: {
  groupLabel: string;
  rows: RowData[];
  visibleFields: FieldDefinition[];
  selectedRowIds: string[];
  rowActionMenuId: string | null;
  setRowActionMenuId: (id: string | null) => void;
  toggleRowSelection: (id: string) => void;
  renderInlineEditor: (row: RowData, field: FieldDefinition) => ReactNode;
  openRecord: (row: RowData) => void;
  editRecord: (row: RowData) => void;
  duplicateRecord: (row: RowData) => void;
  deleteRecord: (row: RowData) => void;
}) {
  return (
    <>
      {groupLabel ? (
        <tr>
          <td colSpan={visibleFields.length + 2} style={groupRowStyle}>
            {groupLabel} · {rows.length}
          </td>
        </tr>
      ) : null}

      {rows.map((row) => (
        <tr key={String(row.id)} style={trStyle}>
          <td style={checkboxCellStyle}>
            <input
              type="checkbox"
              checked={selectedRowIds.includes(String(row.id))}
              onChange={() => toggleRowSelection(String(row.id))}
            />
          </td>

          {visibleFields.map((field) => (
            <td key={field.key} style={tdStyle}>
              {renderInlineEditor(row, field)}
            </td>
          ))}

          <td style={tdStyle}>
            <div style={actionMenuWrapStyle}>
              <button
                type="button"
                onClick={(event) => {
                  event.stopPropagation();
                  setRowActionMenuId(
                    rowActionMenuId === String(row.id) ? null : String(row.id)
                  );
                }}
                style={actionMenuButtonStyle}
              >
                ⋯
              </button>

              {rowActionMenuId === String(row.id) ? (
                <div
                  style={actionDropdownStyle}
                  onClick={(event) => event.stopPropagation()}
                >
                  <button
                    type="button"
                    onClick={() => openRecord(row)}
                    style={actionDropdownButtonStyle}
                  >
                    Open
                  </button>

                  <button
                    type="button"
                    onClick={() => editRecord(row)}
                    style={actionDropdownButtonStyle}
                  >
                    Edit
                  </button>

                  <button
                    type="button"
                    onClick={() => duplicateRecord(row)}
                    style={actionDropdownButtonStyle}
                  >
                    Duplicate
                  </button>

                  <button
                    type="button"
                    onClick={() => deleteRecord(row)}
                    style={actionDropdownDangerButtonStyle}
                  >
                    Delete
                  </button>
                </div>
              ) : null}
            </div>
          </td>
        </tr>
      ))}
    </>
  );
}

function FieldInput({
  field,
  value,
  readOnly,
  employees,
  onChange,
}: {
  field: FieldDefinition;
  value: string | number | boolean | null | undefined;
  readOnly: boolean;
  employees: Employee[];
  onChange: (value: string | number | boolean | null) => void;
}) {
  const optionDescription = getOptionDescription(field.options, value);

  if (readOnly) {
    return (
      <div style={infoBoxStyle}>
        <div style={infoLabelStyle}>{field.label}</div>
        <div style={infoValueStyle}>
          {field.type === "employee"
            ? employees.find((employee) => employee.id === Number(value))?.name ||
              "-"
            : field.type === "boolean"
              ? booleanToYesNo(value)
              : field.type === "date"
                ? formatDate(value)
                : field.type === "time"
                  ? formatTime(value)
                  : String(value || "-")}
        </div>
        <small style={infoHelperStyle}>{field.description}</small>
      </div>
    );
  }

  return (
    <label style={fieldStyle}>
      {field.label}
      <small style={helpTextStyle}>{field.description}</small>

      {field.type === "textarea" ? (
        <textarea
          value={String(value || "")}
          onChange={(event) => onChange(event.target.value)}
          style={textareaStyle}
        />
      ) : null}

      {field.type === "select" || field.type === "boolean" ? (
        <select
          value={
            field.type === "boolean" ? booleanToYesNo(value) : String(value || "")
          }
          onChange={(event) =>
            field.type === "boolean"
              ? onChange(event.target.value === "Yes")
              : onChange(event.target.value)
          }
          style={inputStyle}
        >
          <option value="">Select</option>
          {(field.type === "boolean" ? yesNoOptions : field.options || []).map(
            (option) => (
              <option
                key={option.value}
                value={option.value}
                title={option.description}
              >
                {option.label}
              </option>
            )
          )}
        </select>
      ) : null}

      {field.type === "employee" ? (
        <select
          value={String(value || "")}
          onChange={(event) =>
            onChange(event.target.value ? Number(event.target.value) : null)
          }
          style={inputStyle}
        >
          <option value="">Select employee</option>
          {employees.map((employee) => (
            <option key={employee.id} value={employee.id}>
              {employee.name || `Employee #${employee.id}`}
            </option>
          ))}
        </select>
      ) : null}

      {["text", "date", "time", "number", "url"].includes(field.type) ? (
        <input
          type={
            field.type === "number"
              ? "number"
              : field.type === "date"
                ? "date"
                : field.type === "time"
                  ? "time"
                  : field.type === "url"
                    ? "url"
                    : "text"
          }
          value={String(value || "")}
          onChange={(event) =>
            field.type === "number"
              ? onChange(event.target.value ? Number(event.target.value) : null)
              : onChange(event.target.value)
          }
          style={inputStyle}
        />
      ) : null}

      {optionDescription ? (
        <small style={optionHelpStyle}>{optionDescription}</small>
      ) : null}
    </label>
  );
}

function FieldSelect({
  label,
  value,
  options,
  onChange,
}: {
  label: string;
  value: string;
  options: string[];
  onChange: (value: string) => void;
}) {
  return (
    <label style={fieldStyle}>
      {label}
      <small style={helpTextStyle}>Choose the field to use.</small>
      <select
        value={value}
        onChange={(event) => onChange(event.target.value)}
        style={inputStyle}
      >
        {options.map((key) => {
          const field = getField(key);

          return (
            <option key={key} value={key}>
              {field?.label || key}
            </option>
          );
        })}
      </select>
    </label>
  );
}
