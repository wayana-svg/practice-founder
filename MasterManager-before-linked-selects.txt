function escapeCsvValue(value: string | number | boolean | null | undefined) {
  if (value === null || value === undefined) return "";

  const text = String(value);

  if (text.includes(",") || text.includes('"') || text.includes("\n")) {
    return `"${text.replace(/"/g, '""')}"`;
  }

  return text;
}

function parseCsvLine(line: string) {
  const values: string[] = [];
  let current = "";
  let inQuotes = false;

  for (let index = 0; index < line.length; index += 1) {
    const character = line[index];
    const nextCharacter = line[index + 1];

    if (character === '"' && inQuotes && nextCharacter === '"') {
      current += '"';
      index += 1;
      continue;
    }

    if (character === '"') {
      inQuotes = !inQuotes;
      continue;
    }

    if (character === "," && !inQuotes) {
      values.push(current);
      current = "";
      continue;
    }

    current += character;
  }

  values.push(current);

  return values;
}

function parseCsv(text: string) {
  const lines = text
    .split(/\r?\n/)
    .map((line) => line.trimEnd())
    .filter((line) => line.trim() !== "");

  if (lines.length < 2) {
    return {
      headers: [] as string[],
      rows: [] as Record<string, string>[],
    };
  }

  const headers = parseCsvLine(lines[0]).map((header) => header.trim());

  const rows = lines.slice(1).map((line) => {
    const values = parseCsvLine(line);
    const row: Record<string, string> = {};

    headers.forEach((header, index) => {
      row[header] = values[index] || "";
    });

    return row;
  });

  return { headers, rows };
}

  function rowToDraft(row: RowData) {
    const draft: DraftData = {};

    config.fields.forEach((field) => {
      if (field.type === "smart_due_date") {
        draft[field.key] = "";
        return;
      }

      const value = row[field.key];

      if (value === null || value === undefined) {
        draft[field.key] = "";
      } else {
        draft[field.key] = String(value);
      }
    });

    return draft;
  }

  function convertDraftToPayload(draft: DraftData) {
    const payload: RowData = {};

    config.fields.forEach((field) => {
      if (field.showInDatabasePayload === false) return;
      if (field.type === "smart_due_date") return;
      if (field.key === config.primaryKey) return;

      const value = draft[field.key];

      if (value === "" || value === undefined) {
        payload[field.key] = null;
        return;
      }

      if (
        field.type === "number" ||
        field.type === "currency" ||
        field.type === "employee"
      ) {
        payload[field.key] = Number(value);
        return;
      }

      if (field.type === "boolean") {
        payload[field.key] = value === "true";
        return;
      }

      payload[field.key] = value;
    });

    return payload;
  }

  async function bulkDeleteRecords() {
    const selectedRows = getSelectedRows();

    if (selectedRows.length === 0) return;

    const confirmed = window.confirm(
      `Delete ${selectedRows.length} selected record(s)? This cannot be undone.`
    );

    if (!confirmed) return;

    const ids = selectedRows
      .map((row) => row[config.primaryKey])
      .filter((id): id is string | number => typeof id === "string" || typeof id === "number");

    if (ids.length === 0) return;

    setMessage("");
    setErrorMessage("");

    const supabase = createClient();

    const { error } = await supabase
      .from(config.tableName)
      .delete()
      .in(config.primaryKey, ids);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage(`${selectedRows.length} record(s) deleted.`);
    setSelectedRowIds([]);
    await loadRows();
  }

    await loadRows();
  }

  function exportCsv() {
    const fields = getExportFields();
    const headers = fields.map((field) => field.key);
    const csvRows = rows.map((row) =>
      fields.map((field) => escapeCsvValue(row[field.key])).join(",")
    );

    const csv = [headers.join(","), ...csvRows].join("\n");
    const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
    const url = URL.createObjectURL(blob);
    const link = document.createElement("a");

    link.href = url;
    link.download = `${config.tableName}-export.csv`;
    link.click();

    URL.revokeObjectURL(url);
  }

  async function importCsv(file: File) {
    setMessage("");
    setErrorMessage("");

    const text = await file.text();
    const parsed = parseCsv(text);

    if (parsed.rows.length === 0) {
      setErrorMessage("No rows found in the CSV file.");
      return;
    }

    const fields = getExportFields();

    const payloads = parsed.rows.map((csvRow) => {
      const draft: DraftData = {};

      fields.forEach((field) => {
        const matchingHeader =
          parsed.headers.find(
            (header) =>
              header.toLowerCase() === field.key.toLowerCase() ||
              header.toLowerCase() === field.label.toLowerCase()
          ) || "";

        draft[field.key] = matchingHeader ? csvRow[matchingHeader] || "" : "";
      });

      return convertDraftToPayload(draft);
    });

    const supabase = createClient();

    const { error } = await supabase.from(config.tableName).insert(payloads);

    if (error) {
      setErrorMessage(error.message);
      return;
    }

    setMessage(`${payloads.length} record(s) imported.`);
    await loadRows();
  }

    if (!keys) return;

    const currentText = getSelectedCopyFields(
      dueDateDraft[keys.recurrenceFieldsToCopy] || ""
    );

    const current = currentText
      .split(",")
      .map((item) => item.trim())
      .filter(Boolean);

    const next = current.includes(fieldKey)
      ? current.filter((item) => item !== fieldKey)
      : [...current, fieldKey];

    updateDueDateDraft(
      keys.recurrenceFieldsToCopy,
      next.length === 0 ? "__NONE__" : next.join(",")
    );
  }

  function getSmartDueDateTextFromSource(
    source: DraftData | RowData,
    field: ManagerField
  ) {
    const keys = field.smartDueDateKeys;

    if (!keys) return "Set due date";

    const dueDate = source[keys.dueDate];
    const dueTime = source[keys.dueTime];
    const dueDateType = source[keys.dueDateType] || "Standard";
    const recurrenceFrequency = source[keys.recurrenceFrequency];
    const recurrenceInterval = source[keys.recurrenceInterval] || "1";
    const recurrenceMonthlyRule = source[keys.recurrenceMonthlyRule];
    const recurrenceDays = source[keys.recurrenceDays];

    if (!dueDate) return "Set due date";

    const dateText = String(dueDate);
    const timeText = dueTime ? ` at ${String(dueTime).slice(0, 5)}` : "";

    if (dueDateType === "Repeating") {
      let repeatText = "repeats";

      if (recurrenceFrequency === "Custom") {
        repeatText = `repeats every ${recurrenceInterval} ${recurrenceDays || "Weeks"}`;
      } else if (recurrenceFrequency === "Monthly") {
        repeatText = `repeats ${recurrenceMonthlyRule || "Monthly"}`;
      } else if (recurrenceFrequency === "Weekly") {
        repeatText = `repeats weekly on ${recurrenceDays || "Monday"}`;
      } else if (recurrenceFrequency === "Daily") {
        repeatText = `repeats ${recurrenceDays || "Every Day"}`;
      }

      return `${dateText}${timeText} · ${repeatText}`;
    }

    return `${dateText}${timeText}`;
  }

  function openDueDateModal(field: ManagerField, target: DraftTarget) {
    const draft = getDraftByTarget(target);
    const keys = field.smartDueDateKeys;

    if (!keys) return;

    const nextDueDateDraft: DraftData = {
      [keys.dueDateType]: draft[keys.dueDateType] || "Standard",
      [keys.dueDate]: draft[keys.dueDate] || "",
      [keys.dueTime]: draft[keys.dueTime] || "",
      [keys.recurrenceFrequency]: draft[keys.recurrenceFrequency] || "Daily",
      [keys.recurrenceInterval]: draft[keys.recurrenceInterval] || "1",
      [keys.recurrenceDays]: draft[keys.recurrenceDays] || "Every Day",
      [keys.recurrenceMonthlyRule]:
        draft[keys.recurrenceMonthlyRule] || "Same Day of Month",
      [keys.skipWeekends]: draft[keys.skipWeekends] || "false",
      [keys.createNextWhen]: draft[keys.createNextWhen] || "Complete",
      [keys.recurrenceEndDate]: draft[keys.recurrenceEndDate] || "",
      [keys.nextTaskStatus]: draft[keys.nextTaskStatus] || "To Do",
      [keys.recurrenceFieldsToCopy]:
        draft[keys.recurrenceFieldsToCopy] || getDefaultCopyFieldKeys(),
    };

    setDueDateDraft(nextDueDateDraft);

    const selectedDate = nextDueDateDraft[keys.dueDate];

    if (selectedDate) {
      setCalendarMonth(new Date(`${selectedDate}T12:00:00`));
    } else {
      setCalendarMonth(new Date());
    }

    setDueDateModal({
      open: true,
      target,
      field,
    });
  }

  function saveDueDateModal() {
    if (!dueDateModal.field?.smartDueDateKeys) return;

    const keys = dueDateModal.field.smartDueDateKeys;
    const finalDraft = { ...dueDateDraft };

    if (finalDraft[keys.dueDateType] === "Repeating") {
      if (!finalDraft[keys.dueDate]) {
        setErrorMessage("Choose a start date for the repeating task.");
        return;
      }

      if (!finalDraft[keys.recurrenceFrequency]) {
        finalDraft[keys.recurrenceFrequency] = "Daily";
      }

      if (!finalDraft[keys.recurrenceInterval]) {
        finalDraft[keys.recurrenceInterval] = "1";
      }

      if (!finalDraft[keys.createNextWhen]) {
        finalDraft[keys.createNextWhen] = "Complete";
      }

      if (!finalDraft[keys.nextTaskStatus]) {
        finalDraft[keys.nextTaskStatus] = "To Do";
      }
    }

    updateDraftTarget(dueDateModal.target, finalDraft);
    closeDueDateModal();
  }

  function formatDateForCalendar(date: Date) {
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, "0");
    const day = String(date.getDate()).padStart(2, "0");

    return `${year}-${month}-${day}`;
  }

  function getCalendarDays() {
    const year = calendarMonth.getFullYear();
    const month = calendarMonth.getMonth();
    const firstDay = new Date(year, month, 1);
    const startDay = firstDay.getDay();
    const totalDays = new Date(year, month + 1, 0).getDate();

    const blanks = Array.from({ length: startDay }, () => null);
    const days = Array.from({ length: totalDays }, (_, index) => index + 1);

    return [...blanks, ...days];
  }

      config: field.linkedRecord,
      record: (data as RowData) || null,
      loading: false,
      error: "",
    });
  }

  function formatValue(value: string | number | boolean | null, field?: ManagerField) {
    if (value === null || value === undefined || value === "") return "—";

    if (field?.type === "currency") return `$${Number(value || 0).toFixed(2)}`;

    if (field?.type === "date") {
      const text = String(value);
      return text.includes("T") ? text.split("T")[0] : text;
    }

    if (field?.type === "employee") return getEmployeeName(value);
    if (field?.type === "boolean") return value ? "Yes" : "No";

    if (field?.type === "url") {
      return (
        <a href={String(value)} target="_blank" rel="noopener noreferrer">
          Open
        </a>
      );
    }

    return String(value);
  }

  function compareRows(a: RowData, b: RowData) {
    const field = getField(sortBy);

    if (!field) return 0;

    const first =
      field.type === "smart_due_date" && field.smartDueDateKeys
        ? a[field.smartDueDateKeys.dueDate]
        : a[field.key];

    const second =
      field.type === "smart_due_date" && field.smartDueDateKeys
        ? b[field.smartDueDateKeys.dueDate]
        : b[field.key];

    let result = 0;

    const customOrder = field.sortOrder || [];

    if (customOrder.length > 0) {
      const firstIndex = customOrder.indexOf(String(first));
      const secondIndex = customOrder.indexOf(String(second));

      result =
        (firstIndex === -1 ? 999 : firstIndex) -
        (secondIndex === -1 ? 999 : secondIndex);
    } else if (field.type === "number" || field.type === "currency") {
      result = Number(second || 0) - Number(first || 0);
    } else {
      result = String(first || "").localeCompare(String(second || ""));
    }

    return sortReverse ? result * -1 : result;
  }

                  onClick={saveDueDateModal}
                  style={saveButtonStyle}
                >
                  Done
                </button>
              </div>
            </div>
          </section>
        </div>
      ) : null}

      {aiPromptOpen && aiPromptIsEnabled(config.aiPromptTooling) ? (
        <div style={modalBackdropStyle} onClick={() => setAiPromptOpen(false)}>
          <section style={aiModalStyle} onClick={(event) => event.stopPropagation()}>

              <div>

                <textarea
                  value={aiDone}

                  style={formTextareaStyle}
                />
              </label>
            </div>

            <div style={aiPreviewStyle}>
              <div style={panelLabelStyle}>DESCRIPTION PREVIEW</div>
              <pre style={aiPreviewTextStyle}>{generatedAiDescription()}</pre>
            </div>

    </main>

  function formatLinkedRecordValue(
    value: string | number | boolean | null,
    type?: "text" | "date" | "currency" | "employee" | "boolean"
  ) {
    if (value === null || value === undefined || value === "") return "—";
    if (type === "currency") return `$${Number(value || 0).toFixed(2)}`;
    if (type === "employee") return getEmployeeName(value);
    if (type === "boolean") return value ? "Yes" : "No";
    if (type === "date") {
      const text = String(value);
      return text.includes("T") ? text.split("T")[0] : text;
    }

    return String(value);
  }
}
